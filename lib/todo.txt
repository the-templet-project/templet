(in templet::acta::message use 'ask','reply','tell';
 in templet::acta::actor use 'task')

(in templet::globj use 'update')

async-await model -- templet::async
(in templet::async use 'task', 'wait')

ask-say model- templet::chat
(in templet::chat use 'ask', 'tell')

(in templet::pfor use 'on_iter','on_save','on_load','on_resize')
